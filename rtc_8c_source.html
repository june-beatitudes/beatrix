11static bool
12bea_month_day_is_valid (struct bea_datetime datetime)
13{
14 // Remember here that `datetime.year` is a `uint8_t` with 0 representing the
15 // year 2000
16 bool is_leap_year = (datetime.year % 4 == 0)
17 && ((datetime.year % 100 != 0) || (datetime.year == 0));
18 enum bea_dotw dotw;
19 if (datetime.year == 0)
20 {
21 // Our upcoming formula breaks for 2000 and 2000 only, let's do something
22 // different. The first day of 2000 was a Saturday, and 2000 was a leap
23 // year
24 uint16_t doty = datetime.day;
25 for (size_t i = 0; i < (size_t)datetime.month; ++i)
26 {
27 doty += BEA_MONTH_LENGTHS_LEAP[i];
28 }
29 dotw = (enum bea_dotw) (((size_t)BEA_DOTW_SATURDAY + doty - 1) % 7);
30 }
31 else
32 {
33 // Use Zeller's congruence. I hate this formula because it's ugly as sin
34 // and I had to steal it from Wikipedia, but it is very efficient.
35 uint8_t year_adj
36 = ((uint8_t)datetime.month < 3) ? datetime.year - 1 : datetime.year;
37 uint8_t month_adj = ((uint8_t)datetime.month < 3)
38 ? (uint8_t)datetime.month + 1
39 : (uint8_t)datetime.month + 13;
40 uint8_t zeller_dotw = ((datetime.day + (13 * (month_adj + 1)) / 5
41 + (year_adj % 100) + (year_adj % 100) / 4
42 + (year_adj / 100) / 4 + (year_adj / 100) * 5)
43 % 7);
44 dotw = (enum bea_dotw) ((zeller_dotw + 5) % 7 + 1);
45 }
46 uint8_t days_in_month
47 = is_leap_year ? BEA_MONTH_LENGTHS_LEAP[(uint8_t)datetime.month]
48 : BEA_MONTH_LENGTHS_NONLEAP[(uint8_t)datetime.month];
49 return (datetime.day <= days_in_month && datetime.dotw == dotw);
50}

123bea_rtc_get_datetime ()
124{
125 struct bea_datetime result;
126 uint32_t *t_reg = BEA_RTC_BASE_ADDR + BEA_RTC_TR_OFFSET; // Time register
127 uint32_t *d_reg = BEA_RTC_BASE_ADDR + BEA_RTC_DR_OFFSET; // Date register
128
129 uint32_t second_integral
130 = (bea_get_reg_bits (t_reg, 3, 0) + 10 * bea_get_reg_bits (t_reg, 6, 4));
131 uint32_t prescaler_sync_factor
132 = bea_get_reg_bits (BEA_RTC_BASE_ADDR + BEA_RTC_PRER_OFFSET, 14, 0);
133 float second_fractional
134 = ((float)prescaler_sync_factor
135 - (float)bea_get_reg_bits (BEA_RTC_BASE_ADDR + BEA_RTC_SSR_OFFSET, 15,
136 0))
137 / ((float)prescaler_sync_factor + 1.0f);
138 result.second = (float)second_integral + second_fractional;
139
140 result.minute = bea_get_reg_bits (t_reg, 11, 8)
141 + 10 * bea_get_reg_bits (t_reg, 14, 12);
142 result.hour = bea_get_reg_bits (t_reg, 19, 16)
143 + 10 * bea_get_reg_bits (t_reg, 21, 20);
144 if (bea_get_reg_bits (t_reg, 22, 22) != 0)
145 {
146 result.hour += 12;
147 }
148 // Extract the date, which is in a similarly God-forsaken format
149 result.day
150 = bea_get_reg_bits (d_reg, 3, 0) + 10 * bea_get_reg_bits (d_reg, 5, 4);
151 result.month
152 = (enum bea_month) (bea_get_reg_bits (d_reg, 11, 8)
153 + 10 * bea_get_reg_bits (d_reg, 12, 12) - 1);
154 result.dotw = (enum bea_dotw) (bea_get_reg_bits (d_reg, 15, 13) - 1);
155 result.year = bea_get_reg_bits (d_reg, 19, 16)
156 + 10 * bea_get_reg_bits (d_reg, 23, 20);
157 return result;
158}

160static bool
161bea_rtc_init_prescaler (enum bea_rtc_clksrc clock_source, uint32_t subsec_freq)
162{
163 float rtc_clock_freq = 32768.0f;
164 switch (clock_source)
165 {
166 case BEA_RTC_CLKSRC_HSE:
167 rtc_clock_freq = 1000000.0f;
168 break;
169 case BEA_RTC_CLKSRC_LSE:
170 rtc_clock_freq = 32768.0f;
171 break;
172 case BEA_RTC_CLKSRC_LSI:
173 rtc_clock_freq = 32000.0f;
174 break;
175 }
176 // Either of these will indicate that the subsecond frequency is just too
177 // fast
178 if ((float)subsec_freq > rtc_clock_freq)
179 {
180 return false;
181 }
182 else if (subsec_freq - 1 >= (1 << 15))
183 {
184 return false;
185 }
186 bea_set_reg_bits (BEA_RTC_BASE_ADDR + BEA_RTC_PRER_OFFSET, 14, 0,
187 subsec_freq - 1);
188 float async_factor_f = (rtc_clock_freq / subsec_freq) - 1.0f;
189 if (async_factor_f >= (float)(1 << 7))
190 {
191 return false;
192 }
193 uint32_t async_factor_u = (uint32_t)lroundf (async_factor_f);
194 bea_set_reg_bits (BEA_RTC_BASE_ADDR + BEA_RTC_PRER_OFFSET, 22, 16,
195 async_factor_u);
196 return true;
197}

233bool
234bea_rtc_initialize (enum bea_rtc_clksrc clock_source,
235 struct bea_datetime start_time, uint32_t subsec_freq)
236{
237 uint32_t *bdcr_reg = BEA_RCC_BASE_ADDR + BEA_RCC_BDCR_OFFSET;
238 // Set PWR_CR flag so we can write to RTC registers at all
239 uint32_t *pwr_reg = BEA_PWR_BASE_ADDR + BEA_PWR_CR1_OFFSET;
240 bea_set_reg_bits (pwr_reg, 8, 8, 1);
241 bea_set_reg_bits (bdcr_reg, 16, 16, 1);
242 bea_set_reg_bits (pwr_reg, 8, 8, 1);
243 bea_set_reg_bits (bdcr_reg, 16, 16, 0);
244 // We need to initialize a different register in the RCC module than other
245 // peripherals. Ain't that a pain?
246 bea_set_reg_bits (bdcr_reg, 15, 15, 1);
247 switch (clock_source)
248 {
249 case BEA_RTC_CLKSRC_HSE:
250 bea_set_reg_bits (bdcr_reg, 9, 8, 0b11);
251 break;
252 case BEA_RTC_CLKSRC_LSE:
253 // Make sure the LSE is turned on
254 bea_set_reg_bits (bdcr_reg, 9, 8, 0b01);
255 bea_set_reg_bits (bdcr_reg, 0, 0, 1);
256 while (bea_get_reg_bits (bdcr_reg, 1, 1) == 0)
257 ;
258 break;
259 case BEA_RTC_CLKSRC_LSI:
260 bea_set_reg_bits (bdcr_reg, 9, 8, 0b10);
261 break;
262 }
263 // Disarm the write protection on the RTC registers
264 uint32_t *wpr_reg = BEA_RTC_BASE_ADDR + BEA_RTC_WPR_OFFSET;
265 bea_set_reg_bits (wpr_reg, 31, 0, 0xCA);
266 bea_set_reg_bits (wpr_reg, 31, 0, 0x53);
267 // Set the INIT flag to enter initialization mode
268 uint32_t *isr_reg = BEA_RTC_BASE_ADDR + BEA_RTC_ISR_OFFSET;
269 bea_set_reg_bits (isr_reg, 7, 7, 1);
270 bea_set_reg_bits (isr_reg, 5, 5, 0);
271 // Poll the INITF flag to wait for initialization mode to begin
272 while (bea_get_reg_bits (isr_reg, 6, 6) == 0)
273 ;
274 // Set the prescaler so we can actually measure things
275 bea_rtc_init_prescaler (clock_source, subsec_freq);
276 // Set the start time
277 bea_rtc_set_datetime (start_time);
278 // Exit initialization mode
279 bea_set_reg_bits (isr_reg, 7, 7, 0);
280 // Wait for everything to go through
281 while (bea_get_reg_bits (isr_reg, 5, 5) == 0)
282 ;
283 // Re-arm the write protection on the RTC registers
284 bea_set_reg_bits (wpr_reg, 31, 0, 0xFF);
285 return true;
286}
